import { OnInit, Input, Directive, ElementRef, Renderer2 } from '@angular/core';
import { Button, ButtonType, BUTTONS, DEFAULT_BUTTON_DEF } from './button';

//
// ────────────────────────────────────────────────────────────────────────────── I ──────────
//   :::::: C C   B U T T O N   D I R E C T I V E : :  :   :    :     :        :          :
// ────────────────────────────────────────────────────────────────────────────────────────
//

@Directive({
  selector: 'button[ccButton], a[ccButton]',
  exportAs: 'ccButton',
  host: {
    '[attr.id]': ' _definition.id ? _definition.id : null',
    '[attr.aria-label]': '_definition.ariaLabel ? _definition.ariaLabel : null',
  },
})
export class ButtonDirective implements OnInit {
  /** The JSON definition used to populate the button with content.  */
  @Input()
  public get definition(): Button {
    return this._definition;
  }
  public set definition(buttonDef: Button) {
    this._definition = buttonDef;
  }
  private _definition: Button;

  /** Retrieve the class name for a button with a fill (default). */
  private _fillStyle(buttonType: ButtonType): string {
    const buttonStyles = BUTTONS.get(buttonType);
    return buttonStyles.fillClassName;
  }

  /** Retrieve the class name for an outlined button. */
  private _outlineStyle(buttonType: ButtonType): string {
    const buttonStyles = BUTTONS.get(buttonType);
    return buttonStyles.outlineClassName;
  }

  /** Return a css-friendly string that represents the button's style */
  private get _cssClassFriendlyName(): string {
    if (!this._definition) {
      return;
    }
    const buttonType = this._definition.type;
    if (this._definition.isOutlined) {
      return this._outlineStyle(buttonType);
    } else {
      return this._fillStyle(buttonType);
    }
  }

  /** Return the host element from Angular's ElementRef API */
  protected _getHostElement() {
    return this._elementRef.nativeElement;
  }

  /**
   * Appends a CSS class to the host element without removing or changing the existing class structure.
   * @param className a CSS-friendly, string representation of an HTML class attribute.
   */
  private _addClassName(className: string) {
    if (!className) {
      return null;
    }
    this._renderer.addClass(this._getHostElement(), className);
  }

  /** Apply focus to the host element programatically */
  public focus(): void {
    this._getHostElement().focus();
  }

  constructor(private _elementRef: ElementRef, private _renderer: Renderer2) {}

  ngOnInit() {
    if (!this._definition) {
      this._definition = DEFAULT_BUTTON_DEF;
    }
    this._addClassName('btn');
    this._addClassName(this._cssClassFriendlyName);
  }
}
