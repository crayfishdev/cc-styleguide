import { Component, OnInit } from '@angular/core';
import { Button, ButtonType } from '../lib/button/button';

@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss'],
})
export class AppComponent implements OnInit {
  public defaultButtonObj: Button = {
    id: 'cc-default-button',
    type: ButtonType.DEFAULT,
    ariaLabel: 'This is a default button. Click It!',
    isOutlined: true,
  };

  public primaryButtonObj: Button = {
    id: 'cc-primary-button',
    type: ButtonType.PRIMARY,
    ariaLabel: 'This is a primary button. Click Me!',
    isOutlined: true,
  };

  public successButtonObj: Button = {
    id: 'cc-success-button',
    type: ButtonType.SUCCESS,
    ariaLabel: 'This is a success button. Click here!',
    isOutlined: true,
  };

  public infoButtonObj: Button = {
    id: 'cc-info-button',
    type: ButtonType.INFO,
    ariaLabel: 'This is a information button. Click here!',
    isOutlined: true,
  };

  public warningButtonObj: Button = {
    id: 'cc-warning-button',
    type: ButtonType.WARNING,
    ariaLabel: 'This is a button to warn you. WARNING!',
    isOutlined: true,
  };

  public dangerButtonObj: Button = {
    id: 'cc-danger-button',
    type: ButtonType.DANGER,
    ariaLabel: 'This is a dangerous button. DO NOT CLICK ME!',
    isOutlined: true,
  };

  public linkButtonObj: Button = {
    id: 'cc-link-button',
    type: ButtonType.LINK,
    ariaLabel: 'This is a link. You can click me!',
    isOutlined: true,
  };

  constructor() {}

  ngOnInit() {}
}
